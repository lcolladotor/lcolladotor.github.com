2.1 When you click the Knit button a document will be generated that includes both content as well as the output of any embedded R code chunks within the document.

You can embed an R code chunk like this:

summary(cars)

Code for creating the vignette

## Create the vignette
library('knitrBootstrap') 

knitrBootstrapFlag <- packageVersion('knitrBootstrap') < '1.0.0'
if(knitrBootstrapFlag) {
    ## CRAN version
    library('knitrBootstrap')
    docName <- '2015-01-01-test-5.Rmd'
    system.time(knit_bootstrap(docName, chooser=c('boot', 'code'), show_code = TRUE))
    unlink('dots.md')
} else {
    ## GitHub version
    library('rmarkdown')
    system.time(render(docName))
}
unlink('dotsRef.bib')
